15  package com.cloudera.gertrude;
16  
17  import com.google.common.base.Optional;
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  
21  import java.io.IOException;
22  
23  /**
24   * Manages loading and deserializing new {@link ExperimentSpace} instances and configuring the
25   * {@link ExperimentHandler} to start using them once they have been verified and validated.
26   *
27   * <p>There are many ways to provide data updates to a running binary, such as monitoring a file
28   * or watching a Zookeepeer node. Clients should configure an instance of this class with the
29   * {@link Experiments} namespace at server startup time, and several implementations are provided
30   * in other framework modules.
31   */
32  public abstract class ExperimentSpaceLoader {
33  
34    private static final Logger log = LoggerFactory.getLogger(ExperimentSpaceLoader.class);
35  
36    private ExperimentHandler handler;
37    private ExperimentSpaceDeserializer deserializer;
38  
39    void initialize(ExperimentHandler handler, ExperimentSpaceDeserializer deserializer) {
40      this.handler = handler;
41      this.deserializer = deserializer;
42    }
43  
44    protected synchronized boolean reload(boolean force) {
45      Optional<ExperimentSpace.Serialized> serialized = getSerialized();
46      if (!serialized.isPresent()) {
47        log.warn("No space returned from experiment space supplier, skipping reload");
48        return false;
49      }
50  
51      if (!force && serialized.get().getVersionIdentifier().equals(handler.getVersionIdentifier())) {
52        log.info("Skipping reload because experiment space versions match: {}", serialized.get().getVersionIdentifier());
53        return false;
54      }
55  
56      Optional<ExperimentSpace> data;
57      try {
58        data = deserializer.deserialize(serialized.get());
59      } catch (IOException e) {
60        log.warn("Unable to reload space", e);
61        return false;
62      }
63  
64      if (data.isPresent()) {
65        handler.update(data.get());
66        return true;
67      } else {
68        log.warn("Deserializer could not convert serialized data to a new ExperimentSpace");
69        return false;
70      }
71    }
72  
73    protected abstract Optional<ExperimentSpace.Serialized> getSerialized();
74  }
