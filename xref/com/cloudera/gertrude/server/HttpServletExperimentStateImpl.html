15  package com.cloudera.gertrude.server;
16  
17  import com.cloudera.gertrude.AbstractExperimentState;
18  import com.google.common.base.Optional;
19  import com.google.common.base.Preconditions;
20  import com.google.common.collect.ImmutableMap;
21  
22  import javax.servlet.http.Cookie;
23  import javax.servlet.http.HttpServletRequest;
24  import java.util.List;
25  import java.util.Map;
26  
27  /**
28   * A simple implementation of {@link AbstractExperimentState} and {@link HttpServletExperimentState} that
29   * diverts input requests based on one or more cookie names that are associated with a {@link HttpServletRequest}.
30   */
31  class HttpServletExperimentStateImpl extends AbstractExperimentState implements HttpServletExperimentState {
32  
33    private final HttpServletRequest request;
34    private final Map<Integer, String> diversionCookies;
35  
36    HttpServletExperimentStateImpl(HttpServletRequest request, List<String> diversionCookies) {
37      this.request = Preconditions.checkNotNull(request);
38      this.diversionCookies = indexDiversionCookies(diversionCookies, request.getCookies());
39    }
40  
41    private static Map<Integer, String> indexDiversionCookies(List<String> diversionCookies, Cookie[] cookies) {
42      ImmutableMap.Builder<Integer, String> b = ImmutableMap.builder();
43      for (int i = 0; i < diversionCookies.size(); i++) {
44        String cookieName = diversionCookies.get(i);
45        for (Cookie c : cookies) {
46          if (cookieName.equals(c.getName())) {
47            b.put(i, c.getValue());
48            break;
49          }
50        }
51      }
52      return b.build();
53    }
54  
55    @Override
56    public Optional<String> getDiversionIdentifier(int diversionId) {
57      return Optional.fromNullable(diversionCookies.get(diversionId));
58    }
59  
60    @Override
61    public HttpServletRequest getRequest() {
62      return request;
63    }
64  }
