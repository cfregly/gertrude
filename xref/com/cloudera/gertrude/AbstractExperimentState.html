15  package com.cloudera.gertrude;
16  
17  import com.google.common.base.Optional;
18  import com.google.common.collect.ImmutableSet;
19  import com.google.common.collect.Maps;
20  import com.google.common.collect.Sets;
21  
22  import java.util.Map;
23  import java.util.Set;
24  
25  /**
26   * Abstract base class that implements the core functionality of the {@link ExperimentState} interface.
27   * Clients should create their own {@code ExperimentState} implementations that use this instance as a base
28   * class, since there are core methods in the framework (such as {@link ExperimentHandler#handle(AbstractExperimentState)}
29   * that expect to operate on a subclass of {@code AbstractExperimentState}. More details on the purpose of this
30   * split are described in the documentation of the {@link ExperimentState} interface.
31   */
32  public abstract class AbstractExperimentState implements ExperimentState {
33  
34    private final Map<ExperimentFlag<?>, Object> valueCache = Maps.newHashMap();
35    private final Set<Integer> experimentIds = Sets.newHashSet();
36    private final long requestTimeMsec = System.currentTimeMillis();
37  
38    private ExperimentFlagSettings flagSettings;
39  
40    @Override
41    public abstract Optional<String> getDiversionIdentifier(int diversionId);
42  
43    @Override
44    public Set<Integer> forceExperimentIds() {
45      return ImmutableSet.of();
46    }
47  
48    @Override
49    public <T> T get(ExperimentFlag<T> flag) {
50      if (valueCache.containsKey(flag)) {
51        return (T) valueCache.get(flag);
52      } else if (flagSettings != null) {
53        FlagValue<T> value = flagSettings.getValue(flag, this);
54        if (value.getCacheLevel() != Condition.CacheLevel.NONE) {
55          valueCache.put(flag, value.getValue());
56        }
57        return value.getValue();
58      } else {
59        return flag.getDefaultValue();
60      }
61    }
62  
63    @Override
64    public int getInt(ExperimentFlag<Long> longFlag) {
65      return get(longFlag).intValue();
66    }
67  
68    @Override
69    public float getFloat(ExperimentFlag<Double> doubleFlag) {
70      return get(doubleFlag).floatValue();
71    }
72  
73    @Override
74    public Set<Integer> getExperimentIds()  {
75      return ImmutableSet.copyOf(experimentIds);
76    }
77  
78    @Override
79    public boolean isDiverted() {
80      return flagSettings != null;
81    }
82  
83    @Override
84    public long getRequestTimeMsec() {
85      return requestTimeMsec;
86    }
87  
88    void setFlagSettings(ExperimentFlagSettings flagSettings) {
89      this.valueCache.clear();
90      this.flagSettings = flagSettings;
91    }
92  
93    void addExperimentId(int experimentId) {
94      this.experimentIds.add(experimentId);
95    }
96  }
