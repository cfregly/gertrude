15  package com.cloudera.gertrude;
16  
17  import com.google.common.base.Optional;
18  
19  import java.util.Set;
20  
21  /**
22   * Container for all of the information about how to divert a request into experiments
23   * and what the values of experiment flags are for the current request.
24   *
25   * <p>Most of the logic for managing experiment ids and flag value calculations is defined in the
26   * {@link AbstractExperimentState} class, which clients should extend in order to provide a
27   * definition of the {@link #getDiversionIdentifier(int)} method that is appropriate for their
28   * application.
29   *
30   * <p>{@code ExperimentState} is an interface so that clients may create re-usable families of
31   * {@link Condition} functions that expect a specific sub-interface of ExperimentState and are
32   * aware of how to process them. For example, there could be a {@code HttpServletRequestExperimentState}
33   * sub-interface that provided a {@code getServletRequest()} method, and an associated set of
34   * {@code Condition} functions that could process the {@code HttpServletRequest} and make decisions
35   * based on its values. Then any client could access and use those {@code Condition} functions simply
36   * by extending the {@code AbstractExperimentState} and implementing the {@code HttpServletRequestExperimentState}
37   * interface, without having to re-write their own {@code Condition} implementations for every application.
38   */
39  public interface ExperimentState {
40  
41    /**
42     * Return the identifier to use for this instance with the diversion criterion that
43     * has the given identifer.
44     *
45     * <p>Clients must override this method in order to specify the identifiers that they
46     * support for a given request, and what the values of those identifiers are for each
47     * request.
48     *
49     * <p>Remember that these identifiers are used for randomly diverting requests into
50     * experiments, so they should be unique for each entity that we are experimenting on.
51     * For example, using the country that a request comes from is a relatively poor choice
52     * of a diversion identifier compared to a random account ID or browser cookie.
53     *
54     * @param diversionId The unique id of the requested diversion criterion
55     * @return an identifier to use for diverting this request into an experiment or domain
56     */
57    Optional<String> getDiversionIdentifier(int diversionId);
58  
59    /**
60     * Indicate a set of experiment IDs that this {@code ExperimentState} should be forced
61     * into by the {@link ExperimentHandler#handle(AbstractExperimentState)} method.
62     *
63     * <p>By default, this method returns an empty set. Subclasses may override this
64     * method to provide clients with a way to force a request to be in a certain experiment
65     * in order to test or debug a scenario.
66     *
67     * @return the ids of the experiments that this request should be in
68     */
69    Set<Integer> forceExperimentIds();
70  
71    /**
72     * Returns the value of the given {@code ExperimentFlag} for this {@code ExperimentState}.
73     *
74     * <p>If the {@link FlagValue} that was calculated for this flag can be cached on a per-request
75     * basis, then this instance will keep the calculated value in a cache so that it is not
76     * re-computed unnecessarily during the same request.
77     *
78     * <p>If this state has not yet been diverted by the {@code ExperimentHandler}, then the
79     * default value of the flag is returned, but not stored in the cache.
80     *
81     * @param flag the flag whose value is returned
82     * @return the value of the flag for this instance
83     */
84    <T> T get(ExperimentFlag<T> flag);
85  
86    /**
87     * A convenience method for acccessing the value of a {@code ExperimentFlag<Long>} as an {@code int}.
88     *
89     * @param longFlag the flag
90     * @return the integer point value of the flag
91     */
92    int getInt(ExperimentFlag<Long> longFlag);
93  
94    /**
95     * A convenience method for acccessing the value of a {@code ExperimentFlag<Double>} as a {@code float}.
96     *
97     * @param doubleFlag the flag
98     * @return the floating point value of the flag
99     */
100   float getFloat(ExperimentFlag<Double> doubleFlag);
101 
102   /**
103    * Returns the integer identifiers for the experiments that this state was diverted into.
104    *
105    * @return the integer identifiers for the experiments that this state was diverted into
106    */
107   Set<Integer> getExperimentIds();
108 
109   /**
110    * Indicates whether or not this {@code ExperimentState} has been passed to {@link ExperimentHandler#handle}.
111    *
112    * @return true if this instance has already been diverted, false otherwise
113    */
114   boolean isDiverted();
115 
116   /**
117    * Returns the time that this {@code ExperimentState} instance was created in milliseconds. Used for
118    * determining whether a particular experiment should be active or inactive for this request.
119    *
120    * @return the time that this object was created in milliseconds
121    */
122   long getRequestTimeMsec();
123 }
