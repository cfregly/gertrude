15  package com.cloudera.gertrude;
16  
17  /**
18   * Defines a strategy for converting the string form of a type back into an instance of that type.
19   */
20  public interface FlagTypeParser<T> {
21  
22    /**
23     * Analyze the given value and return an instance of the type class for this instance.
24     *
25     * @param value the value to parse
26     * @return a valid instance of the type class
27     */
28    T parse(CharSequence value);
29  
30    FlagTypeParser<Boolean> BOOLEAN_PARSER = new FlagTypeParser<Boolean>() {
31      @Override
32      public Boolean parse(CharSequence value) {
33        return Boolean.valueOf(value.toString());
34      }
35  
36      @Override
37      public String toString() {
38        return "bool";
39      }
40    };
41  
42    FlagTypeParser<Long> LONG_PARSER = new FlagTypeParser<Long>() {
43      @Override
44      public Long parse(CharSequence value) {
45        return Long.valueOf(value.toString());
46      }
47  
48      @Override
49      public String toString() {
50        return "long";
51      }
52    };
53  
54    FlagTypeParser<Double> DOUBLE_PARSER = new FlagTypeParser<Double>() {
55      @Override
56      public Double parse(CharSequence value) {
57        return Double.valueOf(value.toString());
58      }
59  
60      @Override
61      public String toString() {
62        return "double";
63      }
64    };
65  
66    FlagTypeParser<String> STRING_PARSER = new FlagTypeParser<String>() {
67      @Override
68      public String parse(CharSequence value) {
69        return value.toString();
70      }
71  
72      @Override
73      public String toString() {
74        return "string";
75      }
76    };
77  
78  }
