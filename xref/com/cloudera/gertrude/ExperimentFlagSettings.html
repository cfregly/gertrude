15  package com.cloudera.gertrude;
16  
17  import com.google.common.base.Preconditions;
18  import com.google.common.collect.ImmutableMap;
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  
22  import java.util.Map;
23  
24  /**
25   * A mapping between experiment flag names and the rules for calculating the value of a flag.
26   *
27   * <p>Clients should usually not interact with this class directly, but rather access its
28   * contents on a per-request basis via the {@link ExperimentState#get} method.
29   */
30  public final class ExperimentFlagSettings {
31  
32    private static final Logger log = LoggerFactory.getLogger(ExperimentFlagSettings.class);
33  
34    private final ExperimentFlagSettings delegate;
35    private final Map<String, ? extends FlagValueCalculator<Object>> entries;
36  
37    ExperimentFlagSettings() {
38      this(ImmutableMap.<String, FlagValueCalculator<Object>>of());
39    }
40  
41    ExperimentFlagSettings(Map<String, ? extends FlagValueCalculator<Object>> entries) {
42      this(null, entries);
43    }
44  
45    ExperimentFlagSettings(
46        ExperimentFlagSettings delegate,
47        Map<String, ? extends FlagValueCalculator<Object>> entries) {
48      this.delegate = delegate;
49      this.entries = Preconditions.checkNotNull(entries);
50    }
51  
52    <T> FlagValue<T> getValue(ExperimentFlag<T> flag, ExperimentState state) {
53      FlagValueCalculator<T> calc = (FlagValueCalculator<T>) entries.get(flag.getName());
54      if (calc == null) {
55        if (delegate == null) {
56          log.warn("No calculator defined for experiment flag: {}", flag);
57          return FlagValue.of(flag.getDefaultValue(), Condition.CacheLevel.RELOAD);
58        } else {
59          return delegate.getValue(flag, state);
60        }
61      } else {
62        return calc.apply(state);
63      }
64    }
65  
66    ExperimentFlagSettings withOverrides(Map<String, ? extends FlagValueCalculator<Object>> overrides) {
67      if (overrides.isEmpty()) {
68        return this;
69      }
70      return new ExperimentFlagSettings(this, overrides);
71    }
72  
73    @Override
74    public boolean equals(Object o) {
75      if (this == o) {
76        return true;
77      }
78      if (o == null || getClass() != o.getClass()) {
79        return false;
80      }
81  
82      ExperimentFlagSettings that = (ExperimentFlagSettings) o;
83  
84      return (delegate == null ? that.delegate == null : delegate.equals(that.delegate)) &&
85          entries.equals(that.entries);
86    }
87  
88    @Override
89    public int hashCode() {
90      int result = delegate != null ? delegate.hashCode() : 0;
91      result = 31 * result + entries.hashCode();
92      return result;
93    }
94  }
