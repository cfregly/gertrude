36  final class LayerBuilder {
37  
38    private final LayerInfo info;
39    private final SetMultimap<String, Integer> flagOverridesByExperiment = HashMultimap.create();
40    private final Map<Integer, ExperimentInfo> experiments = Maps.newHashMap();
41    private final Map<Integer, SegmentInfo> domains = Maps.newHashMap();
42    private final Map<Integer, Map<Integer, Integer>> allocatedBucketsByDiversion = Maps.newHashMap();
43    private final ExperimentSpaceBuilder parent;
44  
45    LayerBuilder(ExperimentSpaceBuilder parent, LayerInfo info) {
46      this.parent = parent;
47      this.info = info;
48    }
49  
50    boolean isLaunchLayer() {
51      return info.isLaunchLayer();
52    }
53  
54    int getDomainId() {
55      return info.getDomainId();
56    }
57  
58    void addDomain(SegmentInfo domain) {
59      this.domains.put(domain.getId(), domain);
60    }
61  
62    void addExperiment(ExperimentInfo info) {
63      this.experiments.put(info.getId(), info);
64    }

66    Layer build(Map<Integer, Segment> finalSegments) {
67      ImmutableMap.Builder<Integer, NavigableMap<Integer, Set<Segment>>> b = ImmutableMap.builder();
68      Set<Integer> allSegmentIds = Sets.newHashSet();
69      for (Map.Entry<Integer, Map<Integer, Integer>> e : allocatedBucketsByDiversion.entrySet()) {
70        int diversionId = e.getKey();
71        NavigableMap<Integer, Set<Segment>> bucketToSegment = Maps.newTreeMap();
72        for (Map.Entry<Integer, Integer> ee : e.getValue().entrySet()) {
73          int bucketId = ee.getKey();
74          int segmentId = ee.getValue();
75          Segment segment = experiments.get(segmentId);
76          if (segment == null) {
77            segment = parent.getDomain(domains.get(segmentId), finalSegments);
78          } else {
79            allSegmentIds.add(segmentId);
80          }
81          Set<Segment> segments = bucketToSegment.get(bucketId);
82          if (segments == null) {
83            segments = Sets.newHashSet();
84            bucketToSegment.put(bucketId, segments);
85          }
86          segments.add(segment);
87          finalSegments.put(segmentId, segment);
88        }
89        b.put(diversionId, bucketToSegment);
90      }
91      // TODO: check the collision properties of faster impls (e.g., MurmurHash3)
92      return new LayerImpl(info, Hashing.md5(), allSegmentIds, b.build(), parent.getRandom());
93    }
94  
95    void allocateBuckets(int segmentId, DiversionCriterion criteria, SortedSet<Integer> buckets)
96        throws ValidationException {
97      if (buckets.first() < 0) {
98        throw new ValidationException("Negative buckets in segment: " + segmentId);
99      }
100     if (buckets.last() >= criteria.getNumBuckets()) {
101       throw new ValidationException("Buckets in segment " + segmentId + " exceeds max buckets for criteria");
102     }
103     Map<Integer, Integer> allocatedBuckets = allocatedBucketsByDiversion.get(criteria.getId());
104     if (allocatedBuckets == null) {
105       allocatedBuckets = Maps.newHashMap();
106       allocatedBucketsByDiversion.put(criteria.getId(), allocatedBuckets);
107     }
108     Set<Integer> conflict = Sets.intersection(buckets, allocatedBuckets.keySet());
109     if (!conflict.isEmpty()) {
110       StringBuilder sb = new StringBuilder("Overlapping buckets for segment ")
111           .append(segmentId)
112           .append(" and segment(s) ");
113       Set<Integer> conflictSegments = Sets.newHashSet();
114       for (Integer bucket : conflict) {
115         conflictSegments.add(allocatedBuckets.get(bucket));
116       }
117       sb.append(conflictSegments).append(" (Buckets: ").append(conflict).append(')');
118       throw new ValidationException(sb.toString());
119     }
120     for (Integer bucket : buckets) {
121       allocatedBuckets.put(bucket, segmentId);
122     }
123   }

125   private Iterable<ExperimentInfo> findExperimentsThatOverride(String name) {
126     return Iterables.transform(flagOverridesByExperiment.get(name), new Function<Integer, ExperimentInfo>() {
127       @Override
128       public ExperimentInfo apply(Integer experimentId) {
129         return experiments.get(experimentId);
130       }
131     });
132   }

134   FlagValueData checkOverrides(int experimentId, Map<String, FlagValueOverride<Object>> overrides)
135       throws ValidationException {
136     ImmutableMap.Builder<String, FlagValueCalculatorImpl<Object>> bb = ImmutableMap.builder();
137     Map<Integer, Map<String, FlagValueCalculatorImpl<Object>>> lb = Maps.newHashMap();
138     for (Map.Entry<String, FlagValueOverride<Object>> e : overrides.entrySet()) {
139       String name = e.getKey();
140       FlagValueOverride<Object> valueOverride = e.getValue();
141       FlagValueCalculatorImpl<Object> base = parent.getFlagDefinition(name);
142       if (base == null) {
143         throw new ValidationException("Experiment " + experimentId + " overrides non-existent flag " + name);
144       }
145       bb.put(name, valueOverride.apply(base));
146       flagOverridesByExperiment.put(name, experimentId);
147 
148       //TODO: experiments can share experiment flags if they are in separate domains within a single layer
149       Set<Integer> layerIds = parent.getFlagLayerIds(name);
150       if (!layerIds.contains(info.getLayerId())) {
151         if (!layerIds.isEmpty()) {
152           if (isLaunchLayer()) {
153             throw new ValidationException(String.format(
154                 "Experiments in multiple launch layers override flag %s: layer %d (experiment %d) and layer(s) %s",
155                 name, info.getLayerId(), experimentId, layerIds.toString()));
156           } else {
157             for (Integer otherLayerId : layerIds) {
158               LayerBuilder other = parent.getLayer(otherLayerId);
159               if (other.isLaunchLayer()) {
160                 for (ExperimentInfo info : other.findExperimentsThatOverride(name)) {
161                   Map<String, FlagValueCalculatorImpl<Object>> llOverride = lb.get(info.getId());
162                   if (llOverride == null) {
163                     llOverride = Maps.newHashMap();
164                     lb.put(info.getId(), llOverride);
165                   }
166                   llOverride.put(name, valueOverride.apply(parent.getFlagFromExperiment(info.getId(), name)));
167                 }
168               } else if (parent.areLayersOverlapping(info.getLayerId(), otherLayerId)) {
169                 throw new ValidationException(String.format(
170                     "Experiments across overlapping layers override flag %s: layer %d (experiment %d) and layer(s) %s",
171                     name, info.getLayerId(), experimentId, layerIds.toString()));
172               }
173             }
174           }
175         }
176         parent.addFlagLayerAssignment(name, info.getLayerId());
177       }
178 
179     }
180     return new FlagValueData(bb.build(), lb);
181   }
