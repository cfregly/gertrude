77      this.avroFileInput = avroFileInput;

80    @Override
81    protected Optional<ExperimentSpace> deserialize(ExperimentSpace.Serialized serialized) throws IOException {
82      ExperimentDeployment merged = null;
83      ExperimentDeployment curr = null;
84      if (avroFileInput) {
85        for (InputSupplier<? extends InputStream> is : serialized.getSerializedData()) {
86          SeekableInput si = new SeekableByteArrayInput(ByteStreams.toByteArray(is));
87          FileReader<ExperimentDeployment> dfr = DataFileReader.openReader(si, reader);
88          while (dfr.hasNext()) {
89            merged = merge(merged, dfr.next(curr));
90          }
91        }
92      } else {
93        BinaryDecoder decoder = null;
94        for (InputSupplier<? extends InputStream> is : serialized.getSerializedData()) {
95          decoder = DecoderFactory.get().binaryDecoder(is.getInput(), decoder);
96          merged = merge(merged, reader.read(curr, decoder));
97        }
98      }
99  
100     return merged == null ? Optional.<ExperimentSpace>absent() :
101         Optional.fromNullable(load(merged, serialized.getVersionIdentifier()));
102   }

104   private static ExperimentDeployment merge(ExperimentDeployment one, ExperimentDeployment two) {
105     if (one == null && two != null) {
106       return two;
107     } else {
108       one.setFlagDefinitions(mergeLists(one.getFlagDefinitions(), two.getFlagDefinitions()));
109       one.setDiversions(mergeLists(one.getDiversions(), two.getDiversions()));
110       one.setLayers(mergeLists(one.getLayers(), two.getLayers()));
111       one.setExperiments(mergeLists(one.getExperiments(), two.getExperiments()));
112       return one;
113     }
114   }
115 
116   private static <S> List<S> mergeLists(List<S> one, List<S> two) {
117     if (one == null) {
118       one = Lists.newArrayList();
119     }
120     if (two != null) {
121       one.addAll(two);
122     }
123     return one;
124   }

126   private static <S> List<S> emptyIfNull(List<S> list) {
127     if (list == null) {
128       return ImmutableList.of();
129     }
130     return list;
131   }

133   // Need to merge any existing deployment configs together before calling this
134   public ExperimentSpace load(ExperimentDeployment deployment, String versionIdentifier) {
135     if (deployment.getFlagDefinitions() == null || deployment.getFlagDefinitions().isEmpty()) {
136       log.error("No flags defined in deployment");
137       return null;
138     }
139 
140     ExperimentSpaceBuilder builder = new ExperimentSpaceBuilder(getExperimentFlags(), new Random());
141     Map<String, FlagTypeParser<Object>> parsers = Maps.newHashMap();
142     for (ExperimentFlagDefinition flagDef : deployment.getFlagDefinitions()) {
143       try {
144         addFlagDefinition(flagDef, parsers, builder);
145       } catch (ValidationException e) {
146         log.error("Experiment flag validation error: {}\nFor input: {}", e, flagDef);
147         return null;
148       }
149     }
150 
151     for (DiversionDefinition divDef : emptyIfNull(deployment.getDiversions())) {
152       try {
153         addDiversionCriterion(divDef, builder);
154       } catch (ValidationException e) {
155         log.error("Diversion criteria validation error: {}\nFor input: {}", e, divDef);
156         return null;
157       }
158     }
159 
160     Set<Integer> configuredLayerIds = Sets.newHashSet();
161     Set<Integer> configuredSegmentId = Sets.newHashSet();
162     configuredSegmentId.add(0); // The default domain
163     List<LayerDefinition> layerDefs = emptyIfNull(deployment.getLayers());
164     while (configuredLayerIds.size() < layerDefs.size()) {
165       boolean layersAdded = false;
166 
167       for (LayerDefinition layerDef : layerDefs) {
168         if (configuredSegmentId.contains(layerDef.getDomainId()) &&
169             !configuredLayerIds.contains(layerDef.getId())) {
170           try {
171             addLayer(layerDef, builder);
172           } catch (ValidationException e) {
173             log.error("Layer validation error: {}\nFor input: {}", e, layerDef);
174             return null;
175           }
176           configuredLayerIds.add(layerDef.getId());
177           layersAdded = true;
178         }
179       }
180 
181       if (!layersAdded) {
182         throw new IllegalStateException("Invalid deployment configuration; infinite loop detected");
183       }
184 
185       for (ExperimentDefinition exptDef : emptyIfNull(deployment.getExperiments())) {
186         if (configuredLayerIds.contains(exptDef.getLayerId()) &&
187             !configuredSegmentId.contains(exptDef.getId())) {
188           try {
189             addExperiment(exptDef, parsers, builder);
190           } catch (ValidationException e) {
191             log.error("Experiment validation error: {}\nFor input: {}", e, exptDef);
192             return null;
193           }
194           configuredSegmentId.add(exptDef.getId());
195         }
196       }
197     }
198 
199     return builder.build(versionIdentifier);
200   }

202   void addFlagDefinition(
203       ExperimentFlagDefinition definition,
204       Map<String, FlagTypeParser<Object>> parsers,
205       ExperimentSpaceBuilder builder) throws ValidationException {
206     String flagName = definition.getName().toString();
207     FlagTypeParser<Object> parser = (FlagTypeParser<Object>) getParser(definition.getFlagType());
208     builder.addFlagDefinition(
209         flagName,
210         parser.parse(definition.getBaseValue()),
211         getModifiers(definition.getModifiers(), parser));
212     parsers.put(flagName, parser);
213   }

215   static void addDiversionCriterion(DiversionDefinition diversion, ExperimentSpaceBuilder builder)
216       throws ValidationException {
217     DiversionCriterion dc = new DiversionCriterion(
218         diversion.getId(),
219         diversion.getNumBuckets(),
220         diversion.getRandom());
221     builder.addDiversionCriterion(dc);
222   }

224   static void addLayer(LayerDefinition layerDefinition, ExperimentSpaceBuilder builder)
225       throws ValidationException {
226     LayerInfo info = LayerInfo.builder(layerDefinition.getId())
227         .domainId(layerDefinition.getDomainId())
228         .launchLayer(layerDefinition.getLaunch())
229         .unbiasedId(layerDefinition.getUnbiasedId())
230         .fixedBiasedId(layerDefinition.getFixedBiasedId())
231         .randomBiasedId(layerDefinition.getRandomBiasedId())
232         .build();
233     builder.addLayer(info);
234   }

236   void addExperiment(ExperimentDefinition exptDef,
237                      Map<String, FlagTypeParser<Object>> parsers,
238                      ExperimentSpaceBuilder builder) throws ValidationException {
239     // Needs to be checked against existing bucket ranges
240     SortedSet<Integer> buckets = getBuckets(exptDef.getBuckets(), exptDef.getBucketRanges());
241     Condition<ExperimentState> condition = getCondition(exptDef.getConditions(), exptDef.getConditionMergeOperator());
242     long startTimeMsec = getTime(exptDef.getStartTimeMsecUtc(), 0);
243     long endTimeMsec = getTime(exptDef.getEndTimeMsecUtc(), Long.MAX_VALUE);
244     long prePeriodTimeMsec = getTime(exptDef.getPrePeriodMsecUtc(), startTimeMsec);
245     long postPeriodTimeMsec = getTime(exptDef.getPostPeriodMsecUtc(), endTimeMsec);
246     validateTimes(startTimeMsec, endTimeMsec, prePeriodTimeMsec, postPeriodTimeMsec);
247     SegmentInfo info = new SegmentInfo(exptDef.getId(), exptDef.getLayerId(), exptDef.getDiversionId(),
248         buckets, condition, startTimeMsec, endTimeMsec, prePeriodTimeMsec, postPeriodTimeMsec);
249     Map<String, FlagValueOverride<Object>> overrides = getOverrides(exptDef.getOverrides(),
250         parsers, exptDef.getId());
251     builder.addExperimentInfo(info, exptDef.getDomain(), overrides);
252   }

254   private void validateTimes(
255       long startTimeMsec,
256       long endTimeMsec,
257       long prePeriodTimeMsec,
258       long postPeriodTimeMsec) throws ValidationException {
259     if (startTimeMsec > endTimeMsec) {
260       throw new ValidationException(String.format("Start time (%d) begins after end time (%d)",
261           startTimeMsec, endTimeMsec));
262     } else if (prePeriodTimeMsec < startTimeMsec) {
263       throw new ValidationException(String.format("Pre-period (%d) ends before start time (%d)",
264           prePeriodTimeMsec, startTimeMsec));
265     } else if (endTimeMsec < postPeriodTimeMsec) {
266       throw new ValidationException(String.format("Post-period (%d) starts after end time (%d)",
267           postPeriodTimeMsec, endTimeMsec));
268     }
269   }

271   private long getTime(Long value, long defaultValue) {
272     return value == null ? defaultValue : value;
273   }

275   static FlagTypeParser<?> getParser(FlagType flagType) throws ValidationException {
276     switch (flagType) {
277       case BOOL:
278         return FlagTypeParser.BOOLEAN_PARSER;
279       case INT:
280         return FlagTypeParser.LONG_PARSER;
281       case DOUBLE:
282         return FlagTypeParser.DOUBLE_PARSER;
283       case STRING:
284         return FlagTypeParser.STRING_PARSER;
285       default:
286         throw new ValidationException("Unknown flag type: " + flagType);
287     }
288   }

290   protected Map<String, FlagValueOverride<Object>> getOverrides(
291       List<OverrideDefinition> overrides,
292       Map<String, FlagTypeParser<Object>> parsers,
293       int experimentId) throws ValidationException {
294     ImmutableMap.Builder<String, FlagValueOverride<Object>> b = ImmutableMap.builder();
295     if (overrides != null) {
296       for (OverrideDefinition definition : overrides) {
297         String flagName = definition.getName().toString();
298         FlagTypeParser<Object> parser = parsers.get(flagName);
299         if (parser == null) {
300           throw new ValidationException(String.format(
301               "Unknown experiment flag %s in experiment %d", flagName, experimentId));
302         }
303         List<Modifier<Object>> mods;
304         try {
305           mods = getModifiers(definition.getModifiers(), parser);
306         } catch (ValidationException e) {
307           throw new ValidationException(String.format(
308               "Invalid modifier in overrides for flag %s in experiment %d", flagName, experimentId), e);
309         }
310 
311         FlagValueOverride<Object> flagOverride;
312         switch (definition.getOperator()) {
313           case REPLACE:
314             if (definition.getBaseValue() == null) {
315               throw new ValidationException(String.format(
316                   "REPLACE must have non-null base value for flag %s in experiment %d", flagName, experimentId));
317             }
318             Object baseValue = parser.parse(definition.getBaseValue());
319             flagOverride = FlagValueOverride.createReplace(baseValue, mods);
320             break;
321           case APPEND:
322             flagOverride = FlagValueOverride.createAppend(mods);
323             break;
324           case PREPEND:
325             flagOverride = FlagValueOverride.createPrepend(mods);
326             break;
327           default:
328             throw new ValidationException("Unknown override operator: " + definition.getOperator());
329         }
330         b.put(flagName, flagOverride);
331       }
332     }
333     return b.build();
334   }

336   protected static SortedSet<Integer> getBuckets(List<Integer> buckets, List<BucketRange> bucketRanges) {
337     SortedSet<Integer> ret = Sets.newTreeSet();
338     if (buckets != null) {
339       ret.addAll(buckets);
340     }
341     if (bucketRanges != null) {
342       for (BucketRange br : bucketRanges) {
343         for (int i = br.getStart(); i < br.getEnd(); i++) {
344           ret.add(i);
345         }
346       }
347     }
348     return ret;
349   }

351   protected <T> List<Modifier<T>> getModifiers(List<ModifierDefinition> definitions, FlagTypeParser<T> parser)
352       throws ValidationException{
353     if (definitions == null || definitions.isEmpty()) {
354       return ImmutableList.of();
355     } else {
356       List<Modifier<T>> modifiers = Lists.newArrayListWithExpectedSize(definitions.size());
357       for (ModifierDefinition definition : definitions) {
358         List<Modifier<T>> mods = definition.getModifiers() != null ?
359             getModifiers(definition.getModifiers(), parser) :
360             ImmutableList.<Modifier<T>>of();
361         Condition<ExperimentState> condition =
362             getCondition(definition.getConditions(), definition.getConditionMergeOperator());
363         modifiers.add(new BasicModifier<T>(
364             parser.parse(definition.getValue()),
365             getOperatorFunction(definition.getOperator(), parser),
366             condition,
367             mods));
368       }
369       return modifiers;
370     }
371   }

373   protected Condition<ExperimentState> getCondition(List<ConditionDefinition> definitions, ConditionOperator operator)
374       throws ValidationException {
375     if (definitions == null || definitions.isEmpty()) {
376       return Condition.TRUE;
377     } else {
378       List<Condition<ExperimentState>> conditions = Lists.newArrayList();
379       for (ConditionDefinition definition : definitions) {
380         List<String> args = Lists.transform(emptyIfNull(definition.getArgs()), new Function<Object, String>() {
381           @Override
382           public String apply(Object in) {
383             return in.toString();
384           }
385         });
386         Condition<ExperimentState> c = getConditionFactory().create(definition.getName().toString());
387         if (c != null) {
388           try {
389             c.initialize(args);
390           } catch (Exception e) {
391             throw new ValidationException("Exception initializing condition \"" + definition.getName() + '"', e);
392           }
393           if (definition.getNegate() != null && definition.getNegate()) {
394             c = BooleanConditions.not(c);
395           }
396           conditions.add(c);
397         } else {
398           throw new ValidationException("Unknown condition function name \"" + definition.getName() + '"');
399         }
400       }
401       // May want to cache the c + args mappings to the instances...
402       if (conditions.size() == 1) {
403         return conditions.get(0);
404       }
405       if (operator == null || operator == ConditionOperator.AND) {
406         return BooleanConditions.and(conditions);
407       }
408       if (operator == ConditionOperator.OR) {
409         return BooleanConditions.or(conditions);
410       }
411       throw new ValidationException("Unknown condition operator: " + operator);
412     }
413   }

415   protected static <T> AssociativeOperator<T> getOperatorFunction(
416       ModifierOperator operator,
417       FlagTypeParser<T> parser) {
418     return AssociativeOperator.get(operator.name(), parser);
