41  final class ExperimentSpaceBuilder {
42  
43    private static final Logger log = LoggerFactory.getLogger(ExperimentSpaceBuilder.class);
44  
45    private final Map<String, FlagValueCalculatorImpl<Object>> flagDefinitions = Maps.newHashMap();
46    private final Map<Integer, SegmentInfo> allSegments = Maps.newHashMap();
47    private final SortedMap<Integer, DiversionCriterion> diversionCriteria = Maps.newTreeMap();
48    private final Map<Integer, LayerBuilder> layers = Maps.newHashMap();
49    private final Multimap<Integer, LayerBuilder> layersByDomain = HashMultimap.create();
50    private final SetMultimap<String, Integer> flagLayerAssignments = HashMultimap.create();
51    private final Map<Integer, Map<String, FlagValueCalculatorImpl<Object>>> baseOverridesBySegment = Maps.newHashMap();
52  
53    private final Map<String, ExperimentFlag<?>> experimentFlags;
54    private final Random random;
55  
56    ExperimentSpaceBuilder(Map<String, ExperimentFlag<?>> experimentFlags, Random random) {
57      this.experimentFlags = experimentFlags;
58      this.random = random;
59    }
60  
61    void addFlagDefinition(String name, Object baseValue, List<Modifier<Object>> mods) throws ValidationException {
62      // Ensure that this flag hasn't been defined yet.
63      if (flagDefinitions.containsKey(name)) {
64        throw new ValidationException("Cannot re-define experiment flag \"" + name + '"');
65      }
66  
67      flagDefinitions.put(name, new FlagValueCalculatorImpl<Object>(baseValue, mods));
68  
69      // Second, check compiled flag definition if it exists.
70      ExperimentFlag<?> compiledDef = experimentFlags.get(name);
71      if (compiledDef != null) {
72        // Compare the types of the compiled and external values
73        if (!compiledDef.getDefaultValue().getClass().equals(baseValue.getClass())) {
74          throw new ValidationException(String.format(
75              "Compiled type does not match external type for flag \"%s\" (%s vs. %s)", name,
76              compiledDef.getDefaultValue().getClass().getSimpleName(),
77              baseValue.getClass().getSimpleName()));
78        }
79      } else {
80        log.warn("Experiment flag \"{}\" has not been defined in the compiled environment", name);
81      }
82    }
83  
84    void addDiversionCriterion(DiversionCriterion criteria) throws ValidationException {
85      if (diversionCriteria.containsKey(criteria.getId())) {
86        throw new ValidationException("Cannot re-define diversion criteria: " + criteria.getId());
87      }
88      diversionCriteria.put(criteria.getId(), criteria);
89    }
90  
91    void addLayer(LayerInfo info) throws ValidationException {
92      if (layers.containsKey(info.getLayerId())) {
93        throw new ValidationException("Cannot re-define layer ID: " + info.getLayerId());
94      }
95      if (info.getDomainId() > 0 && info.isLaunchLayer()) {
96        throw new ValidationException(
97            "Launch layer " + info.getLayerId() + " can only be defined in the default domain (domain_id = 0)");
98      }
99      LayerBuilder lb = new LayerBuilder(this, info);
100     layers.put(info.getLayerId(), lb);
101     if (info.getDomainId() > 0) {
102       layersByDomain.put(info.getDomainId(), lb);
103     }
104   }
105 
106   void addExperimentInfo(SegmentInfo info, boolean domain, Map<String, FlagValueOverride<Object>> overrides)
107       throws ValidationException {
108     LayerBuilder layerBuilder = checkSegment(info);
109     allSegments.put(info.getId(), info);
110     if (domain) {
111       layerBuilder.addDomain(info);
112     } else {
113       FlagValueData flagValueData = layerBuilder.checkOverrides(info.getId(), overrides);
114       ExperimentInfo experimentInfo = new ExperimentInfo(info,
115                                                          flagValueData.getBaseOverrides(),
116                                                          flagValueData.getLaunchOverrides());
117       layerBuilder.addExperiment(experimentInfo);
118       baseOverridesBySegment.put(info.getId(), flagValueData.getBaseOverrides());
119     }
120   }
121 
122   ExperimentSpace build(String versionIdentifier) {
123     List<Layer> ret = Lists.newArrayList();
124     Map<Integer, Segment> finalSegments = Maps.newHashMap();
125     for (LayerBuilder lb : layers.values()) {
126       if (lb.getDomainId() == 0) {
127         ret.add(lb.build(finalSegments));
128       }
129     }
130 
131     return new ExperimentSpace(
132         versionIdentifier,
133         flagDefinitions,
134         finalSegments,
135         Lists.newArrayList(diversionCriteria.values()),
136         ret);
137   }
138 
139   private Set<Integer> getLineage(int layerId) {
140     ImmutableSet.Builder<Integer> b = ImmutableSet.builder();
141     b.add(layerId);
142     int domainId = layers.get(layerId).getDomainId();
143     if (domainId > 0) {
144       b.addAll(getLineage(allSegments.get(domainId).getLayerId()));
145     }
146     return b.build();
147   }
148 
149   boolean areLayersOverlapping(int firstLayerId, int secondLayerId) {
150     return Sets.intersection(getLineage(firstLayerId), getLineage(secondLayerId)).isEmpty();
151   }
152 
153   private LayerBuilder checkSegment(SegmentInfo info) throws ValidationException {
154     if (allSegments.containsKey(info.getId())) {
155       throw new ValidationException("Cannot re-define experiment ID: " + info.getId());
156     }
157 
158     LayerBuilder layerBuilder = layers.get(info.getLayerId());
159     if (layerBuilder == null) {
160       throw new ValidationException("Undefined layer ID: " + info.getLayerId());
161     }
162     DiversionCriterion criteria = diversionCriteria.get(info.getDiversionId());
163     if (criteria == null) {
164       throw new ValidationException("Undefined diversion criteria ID: " + info.getDiversionId());
165     }
166 
167     // Make sure we have room for this segment in the current allocation.
168     layerBuilder.allocateBuckets(info.getId(), criteria, info.getBuckets());
169     return layerBuilder;
170   }
171 
172   Random getRandom() {
173     return random;
174   }
175 
176   FlagValueCalculatorImpl<Object> getFlagDefinition(String name) {
177     return flagDefinitions.get(name);
178   }
179 
180   LayerBuilder getLayer(int layerId) {
181     return layers.get(layerId);
182   }
183 
184   Set<Integer> getFlagLayerIds(String name) {
185     return flagLayerAssignments.get(name);
186   }
187 
188   void addFlagLayerAssignment(String name, int layerId) {
189     flagLayerAssignments.put(name, layerId);
190   }
191 
192   Domain getDomain(SegmentInfo info, Map<Integer, Segment> finalSegments) {
193     Set<Layer> layers = Sets.newHashSet();
194     for (LayerBuilder lb : layersByDomain.get(info.getId())) {
195       layers.add(lb.build(finalSegments));
196     }
197     return new Domain(info, layers);
198   }
199 
200   FlagValueCalculatorImpl<Object> getFlagFromExperiment(int id, String name) {
201     return baseOverridesBySegment.get(id).get(name);
202   }
