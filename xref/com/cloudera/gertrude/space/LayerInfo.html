15  package com.cloudera.gertrude.space;
16  
17  public final class LayerInfo {
18    private final int layerId;
19    private final boolean launchLayer;
20    private final int domainId;
21    private final int unbiasedId;
22    private final int fixedBiasedId;
23    private final int randomBiasedId;
24  
25    private LayerInfo(int layerId, boolean launchLayer, int domainId, int unbiasedId, int fixedBiasedId, int randomBiasedId) {
26      this.layerId = layerId;
27      this.launchLayer = launchLayer;
28      this.domainId = domainId;
29      this.unbiasedId = unbiasedId;
30      this.fixedBiasedId = fixedBiasedId;
31      this.randomBiasedId = randomBiasedId;
32    }
33  
34    public int getLayerId() {
35      return layerId;
36    }
37  
38    public boolean isLaunchLayer() {
39      return launchLayer;
40    }
41  
42    public int getDomainId() {
43      return domainId;
44    }
45  
46    public int getUnbiasedId() {
47      return unbiasedId;
48    }
49  
50    public int getFixedBiasedId() {
51      return fixedBiasedId;
52    }
53  
54    public int getRandomBiasedId() {
55      return randomBiasedId;
56    }
57  
58    public static Builder builder(int layerId) {
59      return new Builder(layerId);
60    }
61  
62    public static final class Builder {
63      private final int layerId;
64      private boolean launchLayer;
65      private int domainId;
66      private int unbiasedId;
67      private int fixedBiasedId;
68      private int randomBiasedId;
69  
70      public Builder(int layerId) {
71        this.layerId = layerId;
72      }
73  
74      public Builder launchLayer(boolean launchLayer) {
75        this.launchLayer = launchLayer;
76        return this;
77      }
78  
79      public Builder domainId(int domainId) {
80        this.domainId = domainId;
81        return this;
82      }
83  
84      public Builder unbiasedId(int unbiasedId) {
85        this.unbiasedId = unbiasedId;
86        return this;
87      }
88  
89      public Builder fixedBiasedId(int fixedBiasedId) {
90        this.fixedBiasedId = fixedBiasedId;
91        return this;
92      }
93  
94      public Builder randomBiasedId(int randomBiasedId) {
95        this.randomBiasedId = randomBiasedId;
96        return this;
97      }
98  
99      public LayerInfo build() {
100       return new LayerInfo(layerId, launchLayer, domainId, unbiasedId, fixedBiasedId, randomBiasedId);
101     }
102   }
103 }
