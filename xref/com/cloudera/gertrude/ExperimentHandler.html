15  package com.cloudera.gertrude;
16  
17  import com.codahale.metrics.Meter;
18  import com.codahale.metrics.MetricRegistry;
19  import com.google.common.base.Preconditions;
20  import com.google.common.collect.Sets;
21  
22  import java.util.Set;
23  
24  import static com.codahale.metrics.MetricRegistry.name;
25  
26  /**
27   * Manages experiment diversion for an {@link ExperimentState} based on the data contained in the current
28   * {@link ExperimentSpace}.
29   *
30   * <p>The handler is the link between an {@code ExperimentState} and the currently configured {@link ExperimentSpace}.
31   * As new requests are passed to the {@link #handle(AbstractExperimentState)} method, the {@code ExperimentHandler}
32   * diverts them into experiments that modifies their flag value calculators that are used for the rest of the request
33   * by calling the {@link ExperimentState#get(ExperimentFlag)} methods:
34   *
35   * <pre>   {@code
36   * static ExperimentFlag<Boolean> AWESOME_FEATURE_ON = Experiments.declare("awesome", false);
37   * public void doStuff(Request request) {
38   *    MyExperimentState state = new MyExperimentState(request);
39   *    Experiments.getHandler().handle(state);
40   *    if (state.get(AWESOME_FEATURE_ON)) {
41   *      // enable *AWESOME* feature
42   *    } else {
43   *      // enable feature suggested by product manager
44   *    }
45   * }}</pre>
46   *
47   * <p>New {@code ExperimentSpace} configurations may be updated asynchronously by the {@link ExperimentSpaceLoader}
48   * that is configured for use with this {@code ExperimentHandler} in the {@link Experiments} namespace. Subsequent
49   * calls to the {@link #handle(AbstractExperimentState)} method will use the latest updates to the
50   * {@code ExperimentSpace} for processing requests, and the {@code ExperimentHandler} is thread-safe.
51   */
52  public final class ExperimentHandler {
53  
54    private final MetricRegistry metrics;
55    private final Meter requests;
56  
57    private volatile ExperimentSpace experimentSpace = new ExperimentSpace();
58  
59    ExperimentHandler(MetricRegistry metrics) {
60      this.metrics = Preconditions.checkNotNull(metrics);
61      this.requests = this.metrics.meter(name(ExperimentHandler.class, "requests"));
62    }
63  
64    /**
65     * Diverts the given {@code ExperimentState} into one or more experiments across the
66     * {@link Layer} instances in the current {@link ExperimentSpace}.
67     *
68     * <p>A given {@code ExperimentState} can only be diverted into one experiment per layer;
69     * attempting to re-divert an instance that has already been diverted has no effect on
70     * the flag values or experiment ids associated with the state.
71     *
72     * @param state the request to divert
73     */
74    public void handle(AbstractExperimentState state) {
75      requests.mark();
76  
77      Set<Integer> newExperimentIds = Sets.newHashSet();
78      experimentSpace.diversion(state, newExperimentIds);
79  
80      if (newExperimentIds.isEmpty()) {
81        metrics.meter(name(ExperimentHandler.class, "nodiversion")).mark();
82      } else {
83        for (Integer id : newExperimentIds) {
84          metrics.meter(name(ExperimentHandler.class, String.valueOf(id))).mark();
85          state.addExperimentId(id);
86        }
87      }
88    }
89  
90    /**
91     * Disables the experiment with the given id, preventing it from diverting any traffic.
92     *
93     * @param experimentId id of the experiment to disable
94     * @return true if the experiment was found and disabled, false otherwise
95     */
96    public boolean disable(int experimentId) {
97      return experimentSpace.disable(experimentId);
98    }
99  
100   /**
101    * Returns the version string for the {@code ExperimentSpace} that is currently being used
102    * by this instance to handle requests.
103    *
104    * @return the version string for the current {@code ExperimentSpace}
105    */
106   public String getVersionIdentifier() {
107     return experimentSpace.getVersionIdentifier();
108   }
109 
110   void update(ExperimentSpace experimentSpace) {
111     this.experimentSpace = experimentSpace;
112   }
113 }
