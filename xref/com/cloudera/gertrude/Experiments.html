15  package com.cloudera.gertrude;
16  
17  import com.codahale.metrics.MetricRegistry;
18  import com.google.common.collect.Maps;
19  
20  import java.util.Map;
21  
22  /**
23   * A namespace for declaring, configuring, and accessing the core classes in the Gertrude framework.
24   *
25   * <p>Most clients will get started with Gertrude by declaring {@link ExperimentFlag} instances in
26   * their code that reference parameters that can be calculated by the framework:
27   * <pre>   {@code
28   * ExperimentFlag<Boolean> featureOn = Experiments.declare("feature", false);
29   * ExperimentFlag<Long> upperLimit = Experiments.declare("limit", 1729L);
30   * ExperimentFlag<Double> threshold = Experiments.declare("threshold_for_model", 0.05);
31   * ExperimentFlag<String> background = Experiments.declare("background_color", "white");  }</pre>
32   *
33   * <p>At server startup time, Gertrude needs some additional classes configured so that the framework
34   * can load and process new experiments from external sources:
35   * <ol>
36   *   <li>A {@link ConditionFactory} for mapping from names of {@link Condition} functions to implementations,
37   *   <li>a {@link ExperimentSpaceDeserializer} for processing the serialized form of an {@link ExperimentSpace},
38   *   <li>a {@link ExperimentSpaceLoader} that is configured with the location of the serialized {@link ExperimentSpace},
39   *   <li>and an optional {@link MetricRegistry} for tracking experiment requests and diversions.
40   * </ol>
41   * <p>After these instances are configured, the {@link ExperimentHandler} for the server can be accessed via the
42   * {@link #getHandler()} method and used to divert client requests (represented by a subclass of
43   * {@link AbstractExperimentState}) into experiments that modify parameter values as specified in the configuration
44   * of the {@link ExperimentSpace}.
45   */
46  public final class Experiments {
47    private static final Map<String, ExperimentFlag<?>> EXPERIMENT_FLAGS = Maps.newHashMap();
48  
49    private static ConditionFactory CONDITION_FACTORY;
50    private static ExperimentHandler HANDLER_INSTANCE;
51    private static ExperimentSpaceLoader LOADER_INSTANCE;
52    private static ExperimentSpaceDeserializer DESERIALIZER_INSTANCE;
53    private static MetricRegistry METRIC_REGISTRY_INSTANCE;
54  
55    /**
56     * Declares a boolean experiment flag with the given name and default value.
57     *
58     * @param name the name of the flag
59     * @param defaultValue the default value of the flag
60     * @return the instance of {@code ExperimentFlag} for the name and default value
61     */
62    public static ExperimentFlag<Boolean> declare(String name, boolean defaultValue) {
63      return validate(new ExperimentFlag<Boolean>(name, FlagTypeParser.BOOLEAN_PARSER, defaultValue));
64    }
65  
66    /**
67     * Declares a long experiment flag with the given name and default value.
68     *
69     * @param name the name of the flag
70     * @param defaultValue the default value of the flag
71     * @return the instance of {@code ExperimentFlag} for the name and default value
72     */
73    public static ExperimentFlag<Long> declare(String name, long defaultValue) {
74      return validate(new ExperimentFlag<Long>(name, FlagTypeParser.LONG_PARSER, defaultValue));
75    }
76  
77    /**
78     * Declares a double experiment flag with the given name and default value.
79     *
80     * @param name the name of the flag
81     * @param defaultValue the default value of the flag
82     * @return the instance of {@code ExperimentFlag} for the name and default value
83     */
84    public static ExperimentFlag<Double> declare(String name, double defaultValue) {
85      return validate(new ExperimentFlag<Double>(name, FlagTypeParser.DOUBLE_PARSER, defaultValue));
86    }
87  
88    /**
89     * Declares a string experiment flag with the given name and default value.
90     *
91     * @param name the name of the flag
92     * @param defaultValue the default value of the flag
93     * @return the instance of {@code ExperimentFlag} for the name and default value
94     */
95    public static ExperimentFlag<String> declare(String name, String defaultValue) {
96      return validate(new ExperimentFlag<String>(name, FlagTypeParser.STRING_PARSER, defaultValue));
97    }
98  
99    /**
100    * Register the given {@code ConditionFactory} for creating new {@link Condition} instances from the serialized
101    * {@code ExperimentSpace} configuration.
102    *
103    * <p>If no factory is provided before the call to {@link #getHandler()}, an {@code IllegalStateException}
104    * will be thrown.
105    *
106    * @param factory the configured factory
107    * @return true if the given factory was configured, false if another one had already been configured
108    */
109   public static synchronized boolean registerConditionFactory(ConditionFactory factory) {
110     if (CONDITION_FACTORY != null) {
111       return false;
112     }
113     CONDITION_FACTORY = factory;
114     return true;
115   }
116 
117   /**
118    * Register the given {@code ExperimentSpaceLoader} for retrieving serialized {@link ExperimentSpace}
119    * instances.
120    *
121    * <p>If no loader is provided before the call to {@link #getHandler()}, an {@code IllegalStateException}
122    * will be thrown.
123    *
124    * @param loader the configured loader
125    * @return true if the given loader was configured, false if another one had already been configured
126    */
127   public static synchronized boolean registerLoader(ExperimentSpaceLoader loader) {
128     if (LOADER_INSTANCE != null) {
129       return false;
130     }
131     LOADER_INSTANCE = loader;
132     return true;
133   }
134 
135   /**
136    * Register the given {@code ExperimentSpaceDeserializer} for creating and validating new {@link ExperimentSpace}
137    * instances.
138    *
139    * <p>If no deserializer is provided before the call to {@link #getHandler()}, an {@code IllegalStateException}
140    * will be thrown.
141    *
142    * @param deserializer the configured deserializer
143    * @return true if the given deserializer was configured, false if another one had already been configured
144    */
145   public static synchronized boolean registerDeserializer(ExperimentSpaceDeserializer deserializer) {
146     if (DESERIALIZER_INSTANCE != null) {
147       return false;
148     }
149     DESERIALIZER_INSTANCE = deserializer;
150     return true;
151   }
152 
153   /**
154    * Register the given {@code MetricRegistry} for stat tracking with the framework.
155    *
156    * <p>If no registry is provided before the call to {@link #getHandler()}, the framework will create its
157    * own instance and use that one.
158    *
159    * @param registry the registry instance to use
160    * @return true if the given registry was configured, false if another one had already been configured
161    */
162   public static synchronized boolean registerMetricRegistry(MetricRegistry registry) {
163     if (METRIC_REGISTRY_INSTANCE != null) {
164       return false;
165     }
166     METRIC_REGISTRY_INSTANCE = registry;
167     return true;
168   }
169 
170   public static synchronized ExperimentHandler getHandler() {
171     if (HANDLER_INSTANCE == null) {
172       if (LOADER_INSTANCE == null) {
173         throw new IllegalStateException("No experiment space loader registered for the experiment handler");
174       }
175       if (DESERIALIZER_INSTANCE == null) {
176         throw new IllegalStateException("No experiment space deserializer registered for the experiment handler");
177       }
178       if (CONDITION_FACTORY == null) {
179         throw new IllegalStateException("No condition factory registered for the experiment handler");
180       }
181       if (METRIC_REGISTRY_INSTANCE == null) {
182         METRIC_REGISTRY_INSTANCE = new MetricRegistry();
183       }
184       HANDLER_INSTANCE = new ExperimentHandler(METRIC_REGISTRY_INSTANCE);
185       DESERIALIZER_INSTANCE.initialize(EXPERIMENT_FLAGS, CONDITION_FACTORY);
186       // Initialize and load an initial state
187       LOADER_INSTANCE.initialize(HANDLER_INSTANCE, DESERIALIZER_INSTANCE);
188       LOADER_INSTANCE.reload(true);
189     }
190     return HANDLER_INSTANCE;
191   }
192 
193   private static synchronized <T> ExperimentFlag<T> validate(ExperimentFlag<T> ret) {
194     ExperimentFlag<?> existing = EXPERIMENT_FLAGS.get(ret.getName());
195     if (existing != null) {
196       if (ret.equals(existing)) {
197         return (ExperimentFlag<T>) existing;
198       } else {
199         throw new IllegalStateException("Incompatible flags with the same name: " + existing + " and " + ret);
200       }
201     } else {
202       EXPERIMENT_FLAGS.put(ret.getName(), (ExperimentFlag<Object>) ret);
203       return ret;
204     }
205   }
206 
207   private Experiments() { }
208 }
