15  package com.cloudera.gertrude.deploy;
16  
17  import com.beust.jcommander.Parameter;
18  import com.cloudera.gertrude.ExperimentFlag;
19  import com.cloudera.gertrude.Experiments;
20  import com.google.common.base.Charsets;
21  import com.google.common.collect.ImmutableMap;
22  import com.google.common.io.Files;
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  
26  import java.io.File;
27  import java.io.IOException;
28  import java.util.Locale;
29  import java.util.Map;
30  import java.util.regex.Pattern;
31  
32  public class ExperimentFlagSupport {
33  
34    private static final Logger log = LoggerFactory.getLogger(ExperimentFlagSupport.class);
35    private static final Pattern COMMA = Pattern.compile(",");
36  
37    @Parameter(names = "--flags-file",
38        description = "A CSV file of 'flag_name,flag_type' pairs used for validation.")
39    private String experimentFlagFile;
40  
41    public ExperimentFlagSupport() {}
42  
43    public ExperimentFlagSupport(String experimentFlagFile) {
44      this.experimentFlagFile = experimentFlagFile;
45    }
46  
47    public Map<String, ExperimentFlag<?>> getExperimentFlags() throws IOException {
48      if (experimentFlagFile == null) {
49        return ImmutableMap.of();
50      }
51      ImmutableMap.Builder<String, ExperimentFlag<?>> b = ImmutableMap.builder();
52      for (String line : Files.readLines(new File(experimentFlagFile), Charsets.UTF_8)) {
53        String[] pieces = COMMA.split(line);
54        if (pieces.length == 2) {
55          String flagName = pieces[0];
56          String flagType = pieces[1].toLowerCase(Locale.ENGLISH);
57          if (flagType.startsWith("bool")) {
58            b.put(flagName, Experiments.declare(flagName, false));
59          } else if (flagType.startsWith("int") || "long".equals(flagType)) {
60            b.put(flagName, Experiments.declare(flagName, 0));
61          } else if ("double".equals(flagType)) {
62            b.put(flagName, Experiments.declare(flagName, 0.0));
63          } else if ("string".equals(flagType)) {
64            b.put(flagName, Experiments.declare(flagName, ""));
65          } else {
66            throw new IllegalArgumentException("Unknown flag type: " + flagType);
67          }
68        } else {
69          log.warn("Ignoring invalid line from flag file: {}", line);
70        }
71      }
72      return b.build();
73    }
74  }
