52  public class AvroSupport {
53  
54    @Parameter(names = "--skip-validation", description = "Skip validation checks on input config files.")
55    private boolean skipValidation = false;
56  
57    @Parameter(names = "--timezone-id", description = "ID of the defaullt timezone to use for parsing datetime values")
58    private String timeZoneId = DateTimeZone.getDefault().getID();
59  
60    @ParametersDelegate
61    private ConditionFactorySupport conditionFactorySupport = new ConditionFactorySupport();
62  
63    @ParametersDelegate
64    private ExperimentFlagSupport experimentFlagSupport = new ExperimentFlagSupport();
65  
66    private final DatumWriter<ExperimentDeployment> writer = new SpecificDatumWriter<ExperimentDeployment>(
67        ExperimentDeployment.class);
68  
69    public AvroSupport() {}
70  
71    public AvroSupport(
72        boolean skipValidation,
73        ConditionFactorySupport conditionFactorySupport,
74        ExperimentFlagSupport experimentFlagSupport) {
75      this.skipValidation = skipValidation;
76      this.conditionFactorySupport = conditionFactorySupport;
77      this.experimentFlagSupport = experimentFlagSupport;
78      DateTimeZone.setDefault(DateTimeZone.forID(timeZoneId));
79    }
80  
81    public byte[] toBytes(ExperimentDeployment deployment) throws IOException {
82      ByteArrayOutputStream baos = new ByteArrayOutputStream();
83      BinaryEncoder encoder = EncoderFactory.get().binaryEncoder(baos, null);
84      writer.write(deployment, encoder);
85      encoder.flush();
86      return baos.toByteArray();
87    }
88  
89    public void deploy(ExperimentDeployment deployment, String outputFile) throws IOException {
90      DataFileWriter<ExperimentDeployment> dfw = new DataFileWriter<ExperimentDeployment>(writer)
91          .create(ExperimentDeployment.getClassSchema(), new File(outputFile));
92      dfw.append(deployment);
93      dfw.close();
94    }

96    private static List<? extends Config> getConfigList(Config base, String path, boolean required) {
97      if (!base.hasPath(path)) {
98        if (required) {
99          throw new IllegalArgumentException(String.format("Config %s does not have required path: %s",
100             base.resolve(), path));
101       } else {
102         return ImmutableList.of();
103       }
104     }
105     return base.getConfigList(path);
106   }
107 
108   public ExperimentDeployment createDeployment(Config base) throws IOException {
109     ExperimentDeployment deployment = toExperimentDeployment(base);
110     if (!skipValidation) {
111       AvroExperimentSpaceDeserializer deserializer = new AvroExperimentSpaceDeserializer(false);
112       deserializer.initialize(
113           experimentFlagSupport.getExperimentFlags(),
114           conditionFactorySupport.getConditionFactory());
115       if (deserializer.load(deployment, "VALIDATIION") == null) {
116         return null;
117       }
118     }
119     return deployment;
120   }

122   private static ExperimentDeployment toExperimentDeployment(Config base) {
123     List<ExperimentDefinition> experimentDefinitions = Lists.newArrayList();
124     experimentDefinitions.addAll(getExperiments(getConfigList(base, "EXPERIMENTS", false), false));
125     if (base.hasPath("DOMAINS")) {
126       experimentDefinitions.addAll(getExperiments(getConfigList(base, "DOMAINS", false), true));
127     }
128 
129     return ExperimentDeployment.newBuilder()
130         .setDiversions(getDiversions(getConfigList(base, "DIVERSIONS", true)))
131         .setLayers(getLayers(getConfigList(base, "LAYERS", true)))
132         .setExperiments(experimentDefinitions)
133         .setFlagDefinitions(getFlags(getConfigList(base, "FLAGS", true)))
134         .build();
135   }

137   private static List<ExperimentFlagDefinition> getFlags(List<? extends Config> flagConfig) {
138     return Lists.transform(flagConfig, new Function<Config, ExperimentFlagDefinition>() {
139       @Override
140       public ExperimentFlagDefinition apply(Config input) {
141         String flagName = input.getString("name");
142         if (flagName == null) {
143           throw new IllegalArgumentException("Invalid flag config, no name: " + input.resolve().toString());
144         }
145         return ExperimentFlagDefinition.newBuilder()
146             .setName(flagName)
147             .setBaseValue(input.getString("base-value"))
148             .setDescription(input.getString("description"))
149             .setFlagType(FlagType.valueOf(input.getString("flag-type").toUpperCase(Locale.ENGLISH)))
150             .setModifiers(getModifiers(getConfigList(input, "modifiers", false)))
151             .build();
152       }
153     });
154   }
155 
156   private static List<ModifierDefinition> getModifiers(List<? extends Config> modifierList) {
157     return Lists.transform(modifierList, new Function<Config, ModifierDefinition>() {
158       @Override
159       public ModifierDefinition apply(Config input) {
160         return ModifierDefinition.newBuilder()
161             .setValue(input.getString("value"))
162             .setConditions(getConditions(getConfigList(input, "conditions", false)))
163             .setModifiers(getModifiers(getConfigList(input, "modifiers", false)))
164             .setOperator(ModifierOperator.valueOf(input.getString("operator").toUpperCase(Locale.ENGLISH)))
165             .setConditionMergeOperator(getConditionOperator(input))
166             .build();
167       }
168     });
169   }
170 
171   private static ConditionOperator getConditionOperator(Config input) {
172     if (input.hasPath("condition-merge-operator")) {
173       return ConditionOperator.valueOf(input.getString("condition-merge-operator").toUpperCase(Locale.ENGLISH));
174     }
175     return null;
176   }
177 
178   private static List<ConditionDefinition> getConditions(List<? extends Config> conditionConfig) {
179     return Lists.transform(conditionConfig, new Function<Config, ConditionDefinition>() {
180       @Override
181       public ConditionDefinition apply(Config input) {
182         List<CharSequence> args = Lists.newArrayList();
183         if (input.hasPath("args")) {
184           args.addAll(input.getStringList("args"));
185         }
186         boolean negate = input.hasPath("negate") && input.getBoolean("negate");
187         return ConditionDefinition.newBuilder()
188             .setName(input.getString("name"))
189             .setArgs(args)
190             .setNegate(negate)
191             .build();
192       }
193     });
194   }
195 
196   private static Collection<ExperimentDefinition> getExperiments(List<? extends Config> experimentConfig,
197                                                                  final boolean isDomain) {
198     return Lists.transform(experimentConfig, new Function<Config, ExperimentDefinition>() {
199       @Override
200       public ExperimentDefinition apply(Config input) {
201         return ExperimentDefinition.newBuilder()
202             .setName(input.getString("name"))
203             .setDescription(input.getString("description"))
204             .setOwner(input.getString("owner"))
205             .setId(input.getInt("id"))
206             .setControlId(input.getInt("control-id"))
207             .setLayerId(input.getInt("layer-id"))
208             .setDiversionId(input.getInt("diversion-id"))
209             .setConditions(getConditions(getConfigList(input, "conditions", false)))
210             .setConditionMergeOperator(getConditionOperator(input))
211             .setDomain((input.hasPath("domain") && input.getBoolean("domain")) || isDomain)
212             .setBuckets(input.hasPath("buckets") ? input.getIntList("buckets") : ImmutableList.<Integer>of())
213             .setBucketRanges(getBucketRanges(getConfigList(input, "bucket-ranges", false)))
214             .setStartTimeMsecUtc(getTime(input, "start-time"))
215             .setEndTimeMsecUtc(getTime(input, "end-time"))
216             .setPrePeriodMsecUtc(getTime(input, "preperiod-time"))
217             .setPostPeriodMsecUtc(getTime(input, "postperiod-time"))
218             .setOverrides(getOverrides(getConfigList(input, "overrides", false)))
219             .build();
220       }
221     });
222   }

224   private static Long getTime(Config config, String path) {
225     if (!config.hasPath(path)) {
226       return null;
227     } else {
228       return new DateTime(config.getString(path)).getMillis();
229     }
230   }

245     return Lists.transform(overrideConfig, new Function<Config, OverrideDefinition>() {

258   private static List<LayerDefinition> getLayers(List<? extends Config> layerConfig) {
259     return Lists.transform(layerConfig, new Function<Config, LayerDefinition>() {
260       @Override
261       public LayerDefinition apply(Config input) {
262         return LayerDefinition.newBuilder()

265             .setDomainId(input.hasPath("domain-id") ? input.getInt("domain-id") : 0)
