15  package com.cloudera.gertrude;
16  
17  import com.google.common.base.Preconditions;
18  
19  /**
20   * A rule for diverting an {@link ExperimentState} into a {@link Segment} defined within the
21   * set of {@link Layer} instances in the current {@link ExperimentSpace}.
22   *
23   * <p>For a web application, there may be multiple diversion criteria available, such as
24   * a user's login ID, a cookie stored in a browser, or some other attribute of the request,
25   * such as a search query. The ids of the diversion critera imply a priority ordering of the
26   * criteria (from lowest to highest), so that the diversion rules for a request that contain
27   * multiple identifiers (such as both a login ID and a browser cookie) are unambiguous.
28   *
29   * <p>It is also possible to define a random diversion criterion that is independent of the
30   * request. If one is defined, it should be the lowest priority criteria.
31   *
32   * <p>Each criterion (either fixed or random) must have an associated number of buckets available.
33   * {@code Segment} instances claim buckets for a diversion criterion, and {@code ExperimentState}
34   * instances are assigned to {@code Segment} instances based on how their identifiers are mapped into
35   * buckets.
36   *
37   * <p>See the comments on the {@link ExperimentState#getDiversionIdentifier(int)} method for more
38   * information about working with {@code DiversionCriterion}.
39   */
40  public final class DiversionCriterion implements Comparable<DiversionCriterion> {
41    private final int id;
42    private final int numBuckets;
43    private final boolean random;
44  
45    public DiversionCriterion(int id, int numBuckets, boolean random) {
46      Preconditions.checkArgument(numBuckets > 0, String.format("Non-positive bucket count for diversion id %d: %d",
47          id, numBuckets));
48      this.id = id;
49      this.numBuckets = numBuckets;
50      this.random = random;
51    }
52  
53    /**
54     * Returns the unique id of this diversion criteria that is referenced in
55     * the definition of a {@code Segment}.
56     *
57     * @return the id of this diversion criteria
58     */
59    public int getId() {
60      return id;
61    }
62  
63    /**
64     * Returns the number of buckets for this criteria that may be allocated to {@code Segment} instances.
65     * @return the number of buckets for this criteria
66     */
67    public int getNumBuckets() {
68      return numBuckets;
69    }
70  
71    /**
72     * Returns true if this is a random criteria that is independent of any information about
73     * the request contained in the {@code ExperimentState}.
74     *
75     * @return true for request-independent diversion
76     */
77    public boolean isRandom() {
78      return random;
79    }
80  
81    @Override
82    public int compareTo(DiversionCriterion other) {
83      if (id < other.id) {
84        return -1;
85      }
86      if (id > other.id) {
87        return 1;
88      }
89      return 0;
90    }
91  
92    @Override
93    public boolean equals(Object o) {
94      if (this == o) {
95        return true;
96      }
97      if (o == null || getClass() != o.getClass()) {
98        return false;
99      }
100 
101     DiversionCriterion that = (DiversionCriterion) o;
102 
103     return id == that.id && numBuckets == that.numBuckets && random == that.random;
104   }
105 
106   @Override
107   public int hashCode() {
108     int result = id;
109     result = 31 * result + numBuckets;
110     result = 31 * result + (random ? 1 : 0);
111     return result;
112   }
113 }
