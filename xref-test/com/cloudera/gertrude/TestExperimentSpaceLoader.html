15  package com.cloudera.gertrude;
16  
17  import com.google.common.base.Optional;
18  import com.google.common.collect.ImmutableList;
19  import com.google.common.io.InputSupplier;
20  
21  import java.io.InputStream;
22  import java.util.List;
23  
24  public final class TestExperimentSpaceLoader extends ExperimentSpaceLoader {
25  
26    private String versionIdentifier;
27    private List<InputSupplier<? extends InputStream>> suppliers;
28  
29    public TestExperimentSpaceLoader() {
30      this("");
31    }
32  
33    public TestExperimentSpaceLoader(String versionIdentifier) {
34      this(versionIdentifier, ImmutableList.<InputSupplier<? extends InputStream>>of());
35    }
36  
37    public TestExperimentSpaceLoader(String versionIdentifier, InputSupplier<? extends InputStream> supplier) {
38      this(versionIdentifier, ImmutableList.<InputSupplier<? extends InputStream>>of(supplier));
39    }
40  
41    public TestExperimentSpaceLoader(String versionIdentifier, List<InputSupplier<? extends InputStream>> suppliers) {
42      this.versionIdentifier = versionIdentifier;
43      this.suppliers = suppliers;
44    }
45  
46    public TestExperimentSpaceLoader setVersionIdentifier(String versionIdentifier) {
47      this.versionIdentifier = versionIdentifier;
48      return this;
49    }
50  
51    public TestExperimentSpaceLoader setSupplier(InputSupplier<? extends InputStream> supplier) {
52      return setSuppliers(ImmutableList.<InputSupplier<? extends InputStream>>of(supplier));
53    }
54  
55    public TestExperimentSpaceLoader setSuppliers(List<InputSupplier<? extends InputStream>> suppliers) {
56      this.suppliers = suppliers;
57      return this;
58    }
59  
60    @Override
61    protected Optional<ExperimentSpace.Serialized> getSerialized() {
62      return Optional.of(new ExperimentSpace.Serialized(versionIdentifier, suppliers));
63    }
64  }
