15  package com.cloudera.gertrude;
16  
17  import com.google.common.base.Optional;
18  
19  public final class TestExperimentSpaceDeserializer extends ExperimentSpaceDeserializer {
20  
21    private ExperimentSpace data;
22  
23    public TestExperimentSpaceDeserializer() {
24      this(null);
25    }
26  
27    public TestExperimentSpaceDeserializer(ExperimentSpace data) {
28      this.data = data;
29    }
30  
31    public void setData(ExperimentSpace data) {
32      this.data = data;
33    }
34  
35    @Override
36    protected Optional<ExperimentSpace> deserialize(ExperimentSpace.Serialized serialized) {
37      return data == null ? Optional.of(new ExperimentSpace(serialized.getVersionIdentifier())) : Optional.of(data);
38    }
39  }
