15  package com.cloudera.gertrude;
16  
17  import com.google.common.base.Optional;
18  import com.google.common.collect.Maps;
19  import com.google.common.collect.Sets;
20  
21  import java.util.Collections;
22  import java.util.Map;
23  import java.util.Set;
24  
25  public final class TestExperimentState extends AbstractExperimentState {
26  
27    private final Map<ExperimentFlag<?>, Object> testValues = Maps.newHashMap();
28    private final Set<Integer> forceIds = Sets.newHashSet();
29    private final Map<Integer, String> diversionIds = Maps.newHashMap();
30    private Long requestTimeMsec;
31  
32    @Override
33    public Optional<String> getDiversionIdentifier(int diversionId) {
34      return Optional.fromNullable(diversionIds.get(diversionId));
35    }
36  
37    public TestExperimentState setDiversionIdentifier(int diversionId, String identifier) {
38      this.diversionIds.put(diversionId, identifier);
39      return this;
40    }
41  
42    @Override
43    public Set<Integer> forceExperimentIds() {
44      return forceIds;
45    }
46  
47    public TestExperimentState forceExperimentIds(Integer... ids) {
48      this.forceIds.clear();
49      Collections.addAll(forceIds, ids);
50      return this;
51    }
52  
53    @Override
54    public <T> T get(ExperimentFlag<T> flag) {
55      if (testValues.containsKey(flag)) {
56        return (T) testValues.get(flag);
57      }
58      return super.get(flag);
59    }
60  
61    public <T> TestExperimentState set(ExperimentFlag<T> flag, T value) {
62      testValues.put(flag, value);
63      return this;
64    }
65  
66    public TestExperimentState setRequestTimeMsec(Long requestTimeMsec) {
67      this.requestTimeMsec = requestTimeMsec;
68      return this;
69    }
70  
71    @Override
72    public long getRequestTimeMsec() {
73      return requestTimeMsec == null ? super.getRequestTimeMsec() : requestTimeMsec;
74    }
75  }
