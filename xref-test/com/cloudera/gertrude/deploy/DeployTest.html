15  package com.cloudera.gertrude.deploy;
16  
17  import com.cloudera.gertrude.experiments.avro.ExperimentDeployment;
18  import com.typesafe.config.ConfigFactory;
19  import org.apache.curator.test.TestingServer;
20  import org.junit.Test;
21  
22  import java.io.File;
23  
24  public class DeployTest {
25    @Test
26    public void testAvroFileDeploy() throws Exception {
27      File tmpFile = File.createTempFile("experiments", ".avro");
28      tmpFile.deleteOnExit();
29      AvroSupport avroSupport = new AvroSupport();
30      ExperimentDeployment ed = avroSupport.createDeployment(ConfigFactory.load("gertrude.conf"));
31      avroSupport.deploy(ed, tmpFile.getAbsolutePath());
32    }
33  
34    @Test
35    public void testCuratorDeploy() throws Exception {
36      TestingServer server = new TestingServer();
37      AvroSupport avroSupport = new AvroSupport();
38      CuratorSupport curator = new CuratorSupport(server.getConnectString());
39      ExperimentDeployment ed = avroSupport.createDeployment(ConfigFactory.load("gertrude.conf"));
40      String path = "/some/curator/path";
41      curator.deploy(avroSupport.toBytes(ed), path);
42    }
43  }
